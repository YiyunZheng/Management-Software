CREATE TABLE IF NOT EXISTS Admin (
  Username TEXT NOT NULL UNIQUE ,
  Password TEXT,
  Email TEXT NOT NULL UNIQUE,
  Auth TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS Member (
  Id TEXT PRIMARY KEY,
  Status TEXT,
  Type TEXT,
  Name TEXT,
  Gender TEXT,
  Birth TEXT,
  Email TEXT,
  Phone TEXT,
  Wechat TEXT,
  Country TEXT,
  Address1 TEXT,
  Address2 TEXT,
  City TEXT,
  Postal TEXT,
  StdId TEXT,
  Program TEXT,
  EndDate TEXT,
  Degree TEXT,
  Major1 TEXT,
  Major2 TEXT,
  CareerStatus TEXT,
  Company TEXT,
  Position TEXT,
  SearchAuth TEXT
);

CREATE TABLE IF NOT EXISTS Staff (
  Id TEXT PRIMARY KEY,
  MemId TEXT,
  Dept TEXT,
  Position TEXT,
  Auth TEXT,
  FOREIGN KEY(MemId) REFERENCES Member(Id)
);

CREATE TABLE IF NOT EXISTS Record (
  Id TEXT PRIMARY KEY,
  StfId TEXT,
  Time TEXT,
  Action TEXT,
  FOREIGN KEY(StfId) REFERENCES Staff(Id)
);

CREATE TABLE IF NOT EXISTS OPT (
  Id TEXT PRIMARY KEY,
  Status TEXT,
  MemId TEXT,
  StfId TEXT,
  StartDate TEXT,
  EndData TEXT,
  Title TEXT,
  Position TEXT,
  CardNumber TEXT,
  CardStartDate TEXT,
  CardEndDate TEXT,
  FOREIGN KEY(MemId) REFERENCES Member(Id),
  FOREIGN KEY(StfId) REFERENCES Staff(Id)
);

CREATE TABLE IF NOT EXISTS Log (
  Id TEXT PRIMARY KEY,
  Time TEXT,
  StfId TEXT,
  Action TEXT,
  FOREIGN KEY(StfId) REFERENCES Staff(Id)
);

CREATE TABLE IF NOT EXISTS Orders (
  Id TEXT PRIMARY KEY,
  Status TEXT,
  Time TEXT,
  MemId TEXT,
  StfId TEXT,
  Item TEXT,
  Amount INT,
  Price TEXT,
  Comment TEXT,
  FOREIGN KEY(MemId) REFERENCES Member(Id),
  FOREIGN KEY(StfId) REFERENCES Staff(Id)
);

CREATE TABLE IF NOT EXISTS Treasury (
  Id TEXT PRIMARY KEY,
  StfId TEXT NOT NULL,
  Time TEXT,
  Type TEXT,
  Amount INT,
  Comment TEXT
);
